import { PlusIcon, SearchIcon, XIcon, ChevronDownIcon, ChevronRightIcon, SettingsIcon, UndoIcon, RedoIcon, CalendarIcon, ChevronLeftIcon, ChevronRightIcon as ChevronRight } from "lucide-react";
import React, { useState } from "react";
import { Avatar, AvatarFallback } from "@/components/ui/avatar";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";
import { Card, CardContent } from "@/components/ui/card";
import {
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuTrigger,
} from "@/components/ui/dropdown-menu";
import { Input } from "@/components/ui/input";
import { Switch } from "@/components/ui/switch";
import { Popover, PopoverContent, PopoverTrigger } from "@/components/ui/popover";

export const QworshipHome = (): JSX.Element => {
  const [activeTab, setActiveTab] = useState("Project");
  const [isBuildMode, setIsBuildMode] = useState(true);
  const [selectedDate, setSelectedDate] = useState(new Date(2025, 4, 25)); // May 25, 2025 (Sunday)
  const [currentMonth, setCurrentMonth] = useState(new Date(2025, 4, 1)); // May 2025
  const [isCalendarOpen, setIsCalendarOpen] = useState(false);
  const [isVoiceMenuOpen, setIsVoiceMenuOpen] = useState(false);
  const [isListeningMode, setIsListeningMode] = useState(true);
  const [isSuggestedItemsVisible, setIsSuggestedItemsVisible] = useState(true);
  const [isAddItemVisible, setIsAddItemVisible] = useState(true);
  const [searchQuery, setSearchQuery] = useState("");
  const [isSearchFocused, setIsSearchFocused] = useState(false);
  const [expandedSections, setExpandedSections] = useState({
    "PRE-SERVICE ITEMS": true,
    "WARM - UP": false,
    "SERVICE ITEMS": false,
    "POST SERVICE LOOP": false,
  });

  // Hands-free Bible widget state
  const [isHandsfreeBibleOpen, setIsHandsfreeBibleOpen] = useState(false);
  const [selectedBibleVersion, setSelectedBibleVersion] = useState("KJV");
  const [detectedCommands, setDetectedCommands] = useState("No commands detected");
  const [widgetPosition, setWidgetPosition] = useState({ x: 0, y: 0 });
  const [isDragging, setIsDragging] = useState(false);
  const [dragOffset, setDragOffset] = useState({ x: 0, y: 0 });
  const [hasBeenDragged, setHasBeenDragged] = useState(false);
  const [isSettingsOpen, setIsSettingsOpen] = useState(false);

  // Unified listening state - syncs Button 1 and Hands-free Bible widget
  const toggleListening = (newState: boolean) => {
    setIsListeningMode(newState);
  };

  // Function to toggle widget and reset position
  const toggleHandsfreeBible = () => {
    if (isHandsfreeBibleOpen) {
      setIsHandsfreeBibleOpen(false);
    } else {
      // Reset position to default when opening
      setWidgetPosition({ x: 0, y: 0 });
      setHasBeenDragged(false);
      setIsHandsfreeBibleOpen(true);
    }
  };

  // Drag handlers for the widget
  const handleMouseDown = (e: React.MouseEvent) => {
    if ((e.target as HTMLElement).closest('.no-drag')) return; // Skip dragging for interactive elements
    
    setIsDragging(true);
    const rect = (e.currentTarget as HTMLElement).getBoundingClientRect();
    setDragOffset({
      x: e.clientX - rect.left,
      y: e.clientY - rect.top
    });
  };

  const handleMouseMove = (e: MouseEvent) => {
    if (!isDragging) return;
    
    setWidgetPosition({
      x: e.clientX - dragOffset.x,
      y: e.clientY - dragOffset.y
    });
    setHasBeenDragged(true);
  };

  const handleMouseUp = () => {
    setIsDragging(false);
  };

  // Add and remove mouse event listeners
  React.useEffect(() => {
    if (isDragging) {
      document.addEventListener('mousemove', handleMouseMove);
      document.addEventListener('mouseup', handleMouseUp);
      return () => {
        document.removeEventListener('mousemove', handleMouseMove);
        document.removeEventListener('mouseup', handleMouseUp);
      };
    }
  }, [isDragging, dragOffset]);

  const navItems = [
    { name: "Project", active: true },
    { name: "Insert Item", active: false },
    { name: "Assets", active: false },
    { name: "Songbook", active: false, notification: true },
    { name: "Settings", active: false },
    { name: "Guide", active: false },
    { name: "Help", active: false },
  ];

  const serviceSections = ["PRE-SERVICE ITEMS", "WARM - UP", "SERVICE ITEMS", "POST SERVICE LOOP"];

  const toggleSection = (sectionName: string) => {
    setExpandedSections(prev => ({
      ...prev,
      [sectionName]: !prev[sectionName as keyof typeof prev]
    }));
  };

  // Calendar helper functions
  const formatDate = (date: Date) => {
    const days = ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'];
    const months = ['January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December'];
    
    return `${days[date.getDay()]}, ${months[date.getMonth()]} ${date.getDate()}, ${date.getFullYear()}`;
  };

  const getDaysInMonth = (date: Date) => {
    return new Date(date.getFullYear(), date.getMonth() + 1, 0).getDate();
  };

  const getFirstDayOfMonth = (date: Date) => {
    return new Date(date.getFullYear(), date.getMonth(), 1).getDay();
  };

  const isSunday = (date: Date) => {
    return date.getDay() === 0;
  };

  const isSameDay = (date1: Date, date2: Date) => {
    return date1.getDate() === date2.getDate() && 
           date1.getMonth() === date2.getMonth() && 
           date1.getFullYear() === date2.getFullYear();
  };

  const navigateMonth = (direction: 'prev' | 'next') => {
    const newMonth = new Date(currentMonth);
    if (direction === 'prev') {
      newMonth.setMonth(newMonth.getMonth() - 1);
    } else {
      newMonth.setMonth(newMonth.getMonth() + 1);
    }
    setCurrentMonth(newMonth);
  };

  const selectDate = (day: number) => {
    const newDate = new Date(currentMonth.getFullYear(), currentMonth.getMonth(), day);
    setSelectedDate(newDate);
    setIsCalendarOpen(false);
  };

  const renderCalendar = () => {
    const daysInMonth = getDaysInMonth(currentMonth);
    const firstDay = getFirstDayOfMonth(currentMonth);
    const monthName = currentMonth.toLocaleDateString('en-US', { month: 'long', year: 'numeric' });
    
    // Create array of days
    const days = [];
    
    // Add empty cells for days before the first day of the month
    for (let i = 0; i < firstDay; i++) {
      days.push(<div key={`empty-${i}`} className="w-8 h-8"></div>);
    }
    
    // Add days of the month
    for (let day = 1; day <= daysInMonth; day++) {
      const date = new Date(currentMonth.getFullYear(), currentMonth.getMonth(), day);
      const isSelected = isSameDay(date, selectedDate);
      const isSundayDate = isSunday(date);
      
      days.push(
        <button
          key={day}
          onClick={() => selectDate(day)}
          className={`w-8 h-8 text-sm rounded-full flex items-center justify-center transition-colors ${
            isSelected 
              ? 'bg-[#8356F3] text-white font-semibold' 
              : isSundayDate
                ? 'bg-cyan-400 text-cyan-900 font-medium hover:bg-cyan-300'
                : 'text-white hover:bg-gray-700'
          }`}
        >
          {day}
        </button>
      );
    }
    
    return (
      <div className="p-4 bg-[#1a0f2e] border border-gray-600 rounded-lg shadow-lg">
        {/* Calendar Header */}
        <div className="flex items-center justify-between mb-4">
          <button 
            onClick={() => navigateMonth('prev')}
            className="text-gray-400 hover:text-white p-1"
          >
            <ChevronLeftIcon className="w-5 h-5" />
          </button>
          <h3 className="text-white font-medium">{monthName}</h3>
          <button 
            onClick={() => navigateMonth('next')}
            className="text-gray-400 hover:text-white p-1"
          >
            <ChevronRight className="w-5 h-5" />
          </button>
        </div>
        
        {/* Days of week header */}
        <div className="grid grid-cols-7 gap-1 mb-2">
          {['S', 'M', 'T', 'W', 'T', 'F', 'S'].map((day, index) => (
            <div 
              key={day + index} 
              className={`text-center text-xs font-medium p-1 ${
                index === 0 ? 'text-cyan-400' : 'text-gray-400'
              }`}
            >
              {day}
            </div>
          ))}
        </div>
        
        {/* Calendar Grid */}
        <div className="grid grid-cols-7 gap-1">
          {days}
        </div>
        
        {/* Legend */}
        <div className="mt-3 pt-3 border-t border-gray-600">
          <div className="flex items-center justify-center space-x-4 text-xs">
            <div className="flex items-center space-x-1">
              <div className="w-3 h-3 bg-cyan-400 rounded-full"></div>
              <span className="text-gray-300">Sundays</span>
            </div>
            <div className="flex items-center space-x-1">
              <div className="w-3 h-3 bg-[#8356F3] rounded-full"></div>
              <span className="text-gray-300">Selected</span>
            </div>
          </div>
        </div>
      </div>
    );
  };

  return (
    <div className="bg-[#2a1f4b] w-full h-screen flex flex-col">
      {/* Top Header */}
      <header className="bg-[#1a0f2e] px-4 py-3 flex items-center justify-between border-b border-gray-700">
        <div className="flex items-center space-x-3">
          {/* Qworship Logo */}
          <img 
            src="/qworship-logo.png"
            alt="Qworship Logo"
            className="w-6 h-6"
          />
          <span className="text-white font-medium">New Presentation - Qworship</span>
        </div>
        <div className="flex items-center space-x-4">
          <button className="text-gray-400 hover:text-white">
            <div className="w-5 h-5 border border-gray-400 rounded"></div>
          </button>
          <button className="text-gray-400 hover:text-white">
            <XIcon className="w-5 h-5" />
          </button>
        </div>
      </header>
      {/* Navigation Tabs */}
      <nav className="bg-[#2a1f4b] px-4 py-2 flex items-center justify-between border-b border-gray-700">
        <div className="flex items-center space-x-6">
          {navItems.map((item) => (
            <button
              key={item.name}
              onClick={() => setActiveTab(item.name)}
              className={`text-sm font-medium px-3 py-2 rounded relative ${
                activeTab === item.name
                  ? "text-white bg-[#3d2f5f]"
                  : "text-gray-300 hover:text-white"
              }`}
            >
              {item.name}
              {item.notification && (
                <div className="absolute -top-1 -right-2 w-2 h-2 bg-pink-500 rounded-full animate-beep mt-[11px] mb-[11px] ml-[9px] mr-[9px]"></div>
              )}
            </button>
          ))}
        </div>

        <div className="flex items-center justify-between flex-1">
          {/* Center Section with Undo/Redo and Record */}
          <div className="flex items-center space-x-4 mx-auto">
            {/* Undo/Redo Controls */}
            <div className="flex items-center space-x-3 h-12">
              <div className="w-px h-12 bg-purple-500"></div>
              <div className="flex items-center space-x-2">
                <button className="text-gray-300 hover:text-white p-1 rounded">
                  <UndoIcon className="w-5 h-5" />
                </button>
                <button className="text-gray-300 hover:text-white p-1 rounded">
                  <RedoIcon className="w-5 h-5" />
                </button>
              </div>
              <div className="w-px h-12 bg-purple-500"></div>
            </div>

            {/* Record Button */}
            <button className="bg-white text-purple-800 px-4 py-2 rounded-lg flex items-center space-x-2 text-sm font-medium hover:bg-gray-100">
              <div className="w-3 h-3 bg-red-500 rounded-full"></div>
              <span>Record</span>
            </button>
          </div>

          {/* Right Section */}
          <div className="flex items-center space-x-4">
            {/* Build/Preview Toggle */}
          <div className="flex items-center space-x-3">
            <span className={`text-sm ${isBuildMode ? "text-pink-500" : "text-gray-400"}`}>
              Build
            </span>
            <Switch
              checked={!isBuildMode}
              onCheckedChange={(checked) => setIsBuildMode(!checked)}
              className="data-[state=checked]:bg-purple-600 data-[state=unchecked]:bg-[#34383b]"
            />
            <span className={`text-sm ${!isBuildMode ? "text-white" : "text-gray-400"}`}>
              Preview
            </span>
          </div>

          {/* Listening Status */}
          <div className="bg-[#8356F3] px-4 py-2 rounded-lg flex items-center space-x-2">
            <img 
              src="/qworship-logo.png"
              alt="Qworship Logo"
              className="w-5 h-5"
            />
            <span className="text-white text-sm font-medium">Listening</span>
            <ChevronDownIcon className="w-4 h-4 text-white" />
          </div>

          {/* GO LIVE Button */}
          <Button
            variant="outline"
            className="border-[#8356F3] bg-transparent text-white hover:bg-[#8356F3] px-6"
          >
            GO LIVE
          </Button>

          {/* User Avatar */}
          <Avatar className="w-8 h-8">
            <AvatarFallback className="bg-purple-300 text-purple-800">U</AvatarFallback>
          </Avatar>
          </div>
        </div>
      </nav>

      {/* Main Content Area */}
      <div className="flex flex-1 overflow-hidden">
        {/* Container for secondary nav and content */}
        <div className="flex flex-col w-full">
          {/* Secondary Navigation Bar - positioned to start from Nav Bar 2's right edge */}
          <div className="flex">
            {/* Secondary Navigation Bar - 3/4 width from Nav Bar 2 edge to screen edge */}
            <div className="ml-80 flex-1 h-12 bg-[#3a2f5f] border-b border-gray-600 flex items-center justify-between px-4">
              {/* Left side - Empty space */}
              <div className="flex items-center space-x-4">
              </div>

              {/* Right side - Components */}
              <div className="flex items-center space-x-3">
                {/* Background Dropdown with purple separators */}
                <div className="flex items-center">
                  <div className="w-px h-8 bg-purple-400 mr-1"></div>
                  <DropdownMenu>
                    <DropdownMenuTrigger asChild>
                      <Button variant="ghost" className="text-white hover:bg-[#ffffff20] h-8 px-2">
                        <span className="text-sm">Background</span>
                        <ChevronDownIcon className="w-4 h-4 ml-1" />
                      </Button>
                    </DropdownMenuTrigger>
                    <DropdownMenuContent className="bg-[#1a0f2e] border border-gray-600">
                      <div className="p-2">
                        <div className="text-white text-sm py-1 px-2 hover:bg-gray-700 rounded cursor-pointer">Solid Color</div>
                        <div className="text-white text-sm py-1 px-2 hover:bg-gray-700 rounded cursor-pointer">Gradient</div>
                        <div className="text-white text-sm py-1 px-2 hover:bg-gray-700 rounded cursor-pointer">Image</div>
                        <div className="text-white text-sm py-1 px-2 hover:bg-gray-700 rounded cursor-pointer">Video</div>
                      </div>
                    </DropdownMenuContent>
                  </DropdownMenu>
                  <div className="w-px h-8 bg-purple-400 ml-1"></div>
                </div>

                {/* Image Icon */}
                <Button variant="ghost" className="w-8 h-8 p-0 hover:bg-[#ffffff20] border border-gray-500">
                  <svg className="w-4 h-4 text-white" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                    <rect x="3" y="3" width="18" height="18" rx="2" ry="2"/>
                    <circle cx="8.5" cy="8.5" r="1.5"/>
                    <polyline points="21,15 16,10 5,21"/>
                  </svg>
                </Button>

                {/* Settings Icon */}
                <Button 
                  variant="ghost" 
                  className="w-8 h-8 p-0 hover:bg-[#ffffff20]"
                  onClick={() => setIsSettingsOpen(true)}
                >
                  <SettingsIcon className="w-4 h-4 text-white" />
                </Button>

                {/* Hands Free Bible Button */}
                <Button
                  variant="ghost"
                  onClick={toggleHandsfreeBible}
                  className="text-white hover:bg-[#ffffff20] h-8 px-3 text-sm"
                >
                  Hands Free Bible
                  <ChevronDownIcon className="w-4 h-4 ml-1" />
                </Button>
              </div>
            </div>
          </div>

          {/* Content Area with Sidebar and Main */}
          <div className="flex flex-1">
            {/* Left Sidebar (Nav Bar 2) */}
            <aside className="w-80 bg-[#1a0f2e] border-r border-gray-700 flex flex-col overflow-hidden">
              {/* Getting Started Section */}
              <div className="p-4 border-b border-gray-700">
                <h2 className="text-white text-sm font-medium mb-3">Getting Started</h2>
                
                {/* Date Selector */}
                <div className="mb-4">
                  <label className="text-gray-300 text-xs mb-2 block">Date</label>
                  <Popover open={isCalendarOpen} onOpenChange={setIsCalendarOpen}>
                    <PopoverTrigger asChild>
                      <Button
                        variant="outline"
                        className="w-full justify-start text-left font-normal bg-[#2a1f4b] border-gray-600 text-white hover:bg-[#3a2f5f]"
                      >
                        <CalendarIcon className="mr-2 h-4 w-4" />
                        {formatDate(selectedDate)}
                      </Button>
                    </PopoverTrigger>
                    <PopoverContent className="w-auto p-0 bg-transparent border-none shadow-none" align="start">
                      {renderCalendar()}
                    </PopoverContent>
                  </Popover>
                </div>

                {/* Search */}
                <div className="relative">
                  <label className="text-gray-300 text-xs mb-2 block">Search</label>
                  <div className="relative">
                    <SearchIcon className="absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-400 w-4 h-4" />
                    <Input
                      placeholder="Search slides, Bible verses, songs..."
                      value={searchQuery}
                      onChange={(e) => setSearchQuery(e.target.value)}
                      onFocus={() => setIsSearchFocused(true)}
                      onBlur={() => setTimeout(() => setIsSearchFocused(false), 200)}
                      className="pl-10 pr-10 bg-[#2a1f4b] border-gray-600 text-white placeholder:text-gray-400 focus:border-cyan-400 focus:ring-cyan-400"
                    />
                    {searchQuery && (
                      <button
                        onClick={() => setSearchQuery("")}
                        className="absolute right-3 top-1/2 transform -translate-y-1/2 text-gray-400 hover:text-white"
                      >
                        <XIcon className="w-4 h-4" />
                      </button>
                    )}
                  </div>

                  {/* Search Suggestions Dropdown */}
                  {isSearchFocused && searchQuery && (
                    <div className="absolute top-full left-0 right-0 mt-1 bg-[#1a0f2e] border border-gray-600 rounded-lg shadow-lg z-50 max-h-60 overflow-y-auto">
                      <div className="p-2">
                        <div className="text-cyan-400 text-xs font-medium mb-2">Bible Verses</div>
                        <div className="text-white text-sm py-1 px-2 hover:bg-gray-700 rounded cursor-pointer">John 3:16 - For God so loved the world...</div>
                        <div className="text-white text-sm py-1 px-2 hover:bg-gray-700 rounded cursor-pointer">Psalm 23:1 - The Lord is my shepherd...</div>
                        
                        <div className="text-cyan-400 text-xs font-medium mb-2 mt-3">Worship Songs</div>
                        <div className="text-white text-sm py-1 px-2 hover:bg-gray-700 rounded cursor-pointer">Amazing Grace</div>
                        <div className="text-white text-sm py-1 px-2 hover:bg-gray-700 rounded cursor-pointer">How Great Thou Art</div>
                        
                        <div className="text-cyan-400 text-xs font-medium mb-2 mt-3">Slides</div>
                        <div className="text-white text-sm py-1 px-2 hover:bg-gray-700 rounded cursor-pointer">Welcome Slide</div>
                        <div className="text-white text-sm py-1 px-2 hover:bg-gray-700 rounded cursor-pointer">Offering Time</div>
                      </div>
                    </div>
                  )}
                </div>
              </div>

              {/* Button 1 - Voice Commands */}
              <div className="p-4 border-b border-gray-700">
                <DropdownMenu open={isVoiceMenuOpen} onOpenChange={setIsVoiceMenuOpen}>
                  <DropdownMenuTrigger asChild>
                    <Button 
                      variant="outline" 
                      className="w-full justify-between bg-[#a78bfa] hover:bg-[#9575fa] border-[#7c3aed] text-white h-10"
                    >
                      <span className="text-sm">No Voice commands detected</span>
                      <ChevronDownIcon className="w-4 h-4" />
                    </Button>
                  </DropdownMenuTrigger>
                  <DropdownMenuContent className="w-full bg-[#1a0f2e] border border-gray-600" style={{ width: 'var(--radix-dropdown-menu-trigger-width)' }}>
                    <div className="p-3">
                      <div className="flex items-center justify-between mb-2">
                        <span className="text-white text-sm">Listening Mode</span>
                        <Switch
                          checked={isListeningMode}
                          onCheckedChange={toggleListening}
                          className="data-[state=checked]:bg-[#8356F3] data-[state=unchecked]:bg-gray-600"
                        />
                      </div>
                      <button className="w-full text-left text-white text-sm py-2 px-3 hover:bg-gray-700 rounded">
                        End session
                      </button>
                    </div>
                  </DropdownMenuContent>
                </DropdownMenu>
              </div>

              {/* Button 2 - Suggested Items */}
              {isSuggestedItemsVisible && (
                <div className="p-4 border-b border-gray-700">
                  <Button 
                    variant="outline" 
                    className="w-full justify-between bg-cyan-500 hover:bg-cyan-600 border-cyan-700 text-white h-10 relative"
                  >
                    <span className="text-sm">Suggested Items</span>
                    <button 
                      onClick={(e) => {
                        e.stopPropagation();
                        setIsSuggestedItemsVisible(false);
                      }}
                      className="hover:bg-cyan-700 rounded p-1"
                    >
                      <XIcon className="w-4 h-4" />
                    </button>
                  </Button>
                </div>
              )}

              {/* Service Sections */}
              <div className="flex-1 overflow-y-auto">
                <div className="p-4">
                  {serviceSections.map((section) => (
                    <div key={section} className="mb-4">
                      <button
                        onClick={() => toggleSection(section)}
                        className="w-full flex items-center justify-between p-2 text-white hover:bg-[#ffffff10] rounded transition-colors"
                      >
                        <span className="text-sm font-medium">{section}</span>
                        <ChevronRightIcon 
                          className={`w-4 h-4 transition-transform ${
                            expandedSections[section as keyof typeof expandedSections] 
                              ? 'rotate-90 text-cyan-400' 
                              : 'text-gray-400'
                          }`} 
                        />
                      </button>
                      
                      {expandedSections[section as keyof typeof expandedSections] && (
                        <div className="mt-2">
                          {section === "PRE-SERVICE ITEMS" ? (
                            <div className="space-y-2">
                              <Card className="bg-[#ffffff10] border-l-4 border-l-cyan-400 border-r-0 border-t-0 border-b-0">
                                <CardContent className="p-3">
                                  <div className="flex items-center justify-between">
                                    <div>
                                      <p className="text-white text-sm font-medium">Welcome Slide</p>
                                      <p className="text-gray-400 text-xs">Slide • 2 min</p>
                                    </div>
                                    <div className="flex space-x-1">
                                      <button className="text-gray-400 hover:text-white p-1">
                                        <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M11 5H6a2 2 0 00-2 2v11a2 2 0 002 2h11a2 2 0 002-2v-5m-1.414-9.414a2 2 0 112.828 2.828L11.828 15H9v-2.828l8.586-8.586z" />
                                        </svg>
                                      </button>
                                      <button className="text-gray-400 hover:text-red-400 p-1">
                                        <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16" />
                                        </svg>
                                      </button>
                                    </div>
                                  </div>
                                </CardContent>
                              </Card>
                              <Card className="bg-[#ffffff10] border-l-4 border-l-purple-400 border-r-0 border-t-0 border-b-0">
                                <CardContent className="p-3">
                                  <div className="flex items-center justify-between">
                                    <div>
                                      <p className="text-white text-sm font-medium">Announcements</p>
                                      <p className="text-gray-400 text-xs">Slide • 3 min</p>
                                    </div>
                                    <div className="flex space-x-1">
                                      <button className="text-gray-400 hover:text-white p-1">
                                        <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M11 5H6a2 2 0 00-2 2v11a2 2 0 002 2h11a2 2 0 002-2v-5m-1.414-9.414a2 2 0 112.828 2.828L11.828 15H9v-2.828l8.586-8.586z" />
                                        </svg>
                                      </button>
                                      <button className="text-gray-400 hover:text-red-400 p-1">
                                        <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16" />
                                        </svg>
                                      </button>
                                    </div>
                                  </div>
                                </CardContent>
                              </Card>
                            </div>
                          ) : (
                            <div className="ml-7 py-2">
                              <p className="text-gray-400 text-xs">No items in this section</p>
                            </div>
                          )}
                        </div>
                      ))}
                    </div>
                  ))}
                </div>
              </div>

              {/* Add Item */}
              {isAddItemVisible && (
                <div className="p-4">
                  <Card className="bg-[#ffffff30] border-none">
                    <CardContent className="p-4 text-center relative">
                      <div className="flex items-center justify-center space-x-2">
                        <PlusIcon className="w-5 h-5 text-white opacity-80" />
                        <span className="text-white text-sm opacity-80">Add Item</span>
                      </div>
                      <button 
                        onClick={() => setIsAddItemVisible(false)}
                        className="absolute top-2 right-2 hover:bg-[#ffffff20] rounded p-1 transition-colors"
                      >
                        <XIcon className="w-4 h-4 text-white opacity-60 hover:opacity-100" />
                      </button>
                    </CardContent>
                  </Card>
                </div>
              )}
            </aside>

            {/* Main Presentation Area */}
            <main className="flex-1 bg-[#6b5b8f] p-8 flex items-center justify-center">
              <div className="w-full h-full bg-[#7c6ba0] rounded-lg flex items-center justify-center">
                <div className="text-center">
                  <h3 className="text-white text-xl opacity-60">Presentation Preview</h3>
                  <p className="text-white opacity-40 mt-2">Your content will appear here</p>
                </div>
              </div>
            </main>
          </div>
        </div>

        {/* Bottom Timeline */}
        <div className="h-4 bg-[#2a1f4b] px-4 py-2 flex items-center">
          <div className="w-full max-w-4xl mx-auto h-2 bg-gray-600 rounded-full"></div>
        </div>
      </div>

      {/* Hands-free Bible Widget */}
      {isHandsfreeBibleOpen && (
        <div
          className={`${
            hasBeenDragged ? 'fixed' : 'absolute'
          } bg-[#1a0f2e] border border-gray-600 rounded-lg shadow-xl z-50 w-80`}
          style={{
            ...(hasBeenDragged
              ? { left: widgetPosition.x, top: widgetPosition.y }
              : { right: '1rem', top: '8rem' })
          }}
          onMouseDown={handleMouseDown}
        >
          {/* Widget Header */}
          <div className="p-4 border-b border-gray-600 cursor-move">
            <div className="flex items-center justify-between">
              <div className="flex items-center space-x-2">
                {/* Qworship Logo */}
                <div className="w-6 h-6 bg-pink-500 rounded-full flex items-center justify-center relative">
                  <div className="w-3 h-3 bg-black rounded-full relative">
                    <div className="w-1 h-1 bg-pink-500 rounded-full absolute top-0.5 left-0.5"></div>
                  </div>
                </div>
                <span className="text-white text-sm font-medium">Hands-free Bible</span>
              </div>
              <button
                onClick={toggleHandsfreeBible}
                className="text-gray-400 hover:text-white no-drag"
              >
                <XIcon className="w-4 h-4" />
              </button>
            </div>
          </div>

          {/* Widget Content */}
          <div className="p-4">
            {/* Microphone Section */}
            <div className="flex items-center justify-center mb-4">
              <div className={`w-16 h-16 rounded-full flex items-center justify-center transition-all duration-300 ${
                isListeningMode 
                  ? 'bg-[#8356F3] animate-pulse' 
                  : 'bg-gray-600'
              }`}>
                {isListeningMode ? (
                  <svg className="w-8 h-8 text-white" fill="currentColor" viewBox="0 0 20 20">
                    <path fillRule="evenodd" d="M7 4a3 3 0 016 0v4a3 3 0 11-6 0V4zm4 10.93A7.001 7.001 0 0017 8a1 1 0 10-2 0A5 5 0 015 8a1 1 0 00-2 0 7.001 7.001 0 006 6.93V17H6a1 1 0 100 2h8a1 1 0 100-2h-3v-2.07z" clipRule="evenodd" />
                  </svg>
                ) : (
                  <div className="relative">
                    <svg className="w-8 h-8 text-white" fill="currentColor" viewBox="0 0 20 20">
                      <path fillRule="evenodd" d="M7 4a3 3 0 016 0v4a3 3 0 11-6 0V4zm4 10.93A7.001 7.001 0 0017 8a1 1 0 10-2 0A5 5 0 015 8a1 1 0 00-2 0 7.001 7.001 0 006 6.93V17H6a1 1 0 100 2h8a1 1 0 100-2h-3v-2.07z" clipRule="evenodd" />
                    </div>
                    <div className="absolute inset-0 flex items-center justify-center">
                      <div className="w-10 h-0.5 bg-red-500 rotate-45"></div>
                    </div>
                  </div>
                )}
              </div>
            </div>

            {/* Status Text */}
            <div className="text-center mb-4">
              <p className="text-white text-sm">
                {isListeningMode ? 'Qworship is listening' : 'Qworship is off'}
              </p>
            </div>

            {/* Bible Version Tabs */}
            <div className="flex space-x-1 mb-4 no-drag">
              {['KJV', 'NKJV', 'AMP', 'GN', 'MSG'].map((version) => (
                <button
                  key={version}
                  onClick={() => setSelectedBibleVersion(version)}
                  className={`px-3 py-1 text-xs rounded transition-colors ${
                    selectedBibleVersion === version
                      ? 'bg-[#8356F3] text-white'
                      : 'bg-gray-700 text-gray-300 hover:bg-gray-600'
                  }`}
                >
                  {version}
                </button>
              ))}
            </div>

            {/* Commands Display */}
            <div className="bg-[#2a1f4b] rounded p-3 mb-4">
              <div className="text-gray-300 text-xs mb-1">Commands</div>
              <div className="text-white text-sm">{detectedCommands}</div>
            </div>

            {/* Navigation Controls */}
            <div className="flex items-center justify-between text-sm no-drag">
              <button className="text-gray-400 hover:text-white">← Previous</button>
              <span className="text-gray-300">Verse 1 of 1</span>
              <button className="text-gray-400 hover:text-white">Next →</button>
            </div>

            {/* Settings Button */}
            <div className="mt-4 flex justify-center no-drag">
              <button
                onClick={() => setIsSettingsOpen(true)}
                className="text-gray-400 hover:text-white text-sm flex items-center space-x-1"
              >
                <SettingsIcon className="w-4 h-4" />
                <span>Settings</span>
              </button>
            </div>
          </div>
        </div>
      )}

      {/* Settings Modal */}
      {isSettingsOpen && (
        <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50">
          <div className="bg-[#1a0f2e] border border-gray-600 rounded-lg w-96 max-h-[80vh] overflow-y-auto">
            {/* Settings Header */}
            <div className="p-4 border-b border-gray-600 flex items-center justify-between">
              <h3 className="text-white text-lg font-medium">Settings</h3>
              <button
                onClick={() => setIsSettingsOpen(false)}
                className="text-gray-400 hover:text-white"
              >
                <XIcon className="w-5 h-5" />
              </button>
            </div>

            {/* Settings Content */}
            <div className="p-4 space-y-4">
              {/* Font Type */}
              <div>
                <label className="text-white text-sm font-medium block mb-2">Font Type</label>
                <select className="w-full bg-[#2a1f4b] border border-gray-600 rounded px-3 py-2 text-white">
                  <option>Arial</option>
                  <option>Helvetica</option>
                  <option>Times New Roman</option>
                  <option>Calibri</option>
                </select>
              </div>

              {/* Font Colour */}
              <div>
                <label className="text-white text-sm font-medium block mb-2">Font Colour</label>
                <div className="flex space-x-2">
                  <div className="w-8 h-8 bg-white border border-gray-600 rounded cursor-pointer"></div>
                  <div className="w-8 h-8 bg-black border border-gray-600 rounded cursor-pointer"></div>
                  <div className="w-8 h-8 bg-blue-500 border border-gray-600 rounded cursor-pointer"></div>
                  <div className="w-8 h-8 bg-red-500 border border-gray-600 rounded cursor-pointer"></div>
                </div>
              </div>

              {/* Font Styles */}
              <div>
                <label className="text-white text-sm font-medium block mb-2">Font Styles</label>
                <div className="space-y-2">
                  <label className="flex items-center text-white text-sm">
                    <input type="checkbox" className="mr-2" />
                    Bold
                  </label>
                  <label className="flex items-center text-white text-sm">
                    <input type="checkbox" className="mr-2" />
                    Italic
                  </label>
                  <label className="flex items-center text-white text-sm">
                    <input type="checkbox" className="mr-2" />
                    Underline
                  </label>
                </div>
              </div>

              {/* Background Setting */}
              <div>
                <label className="text-white text-sm font-medium block mb-2">Background Setting</label>
                <select className="w-full bg-[#2a1f4b] border border-gray-600 rounded px-3 py-2 text-white">
                  <option>Solid Color</option>
                  <option>Gradient</option>
                  <option>Image</option>
                  <option>Video</option>
                </select>
              </div>

              {/* Positioning */}
              <div>
                <label className="text-white text-sm font-medium block mb-2">Positioning</label>
                <div className="grid grid-cols-3 gap-2">
                  <button className="bg-[#2a1f4b] border border-gray-600 rounded p-2 text-white text-xs hover:bg-[#3a2f5f]">
                    Top Left
                  </button>
                  <button className="bg-[#2a1f4b] border border-gray-600 rounded p-2 text-white text-xs hover:bg-[#3a2f5f]">
                    Top Center
                  </button>
                  <button className="bg-[#2a1f4b] border border-gray-600 rounded p-2 text-white text-xs hover:bg-[#3a2f5f]">
                    Top Right
                  </button>
                  <button className="bg-[#2a1f4b] border border-gray-600 rounded p-2 text-white text-xs hover:bg-[#3a2f5f]">
                    Center Left
                  </button>
                  <button className="bg-[#8356F3] border border-[#8356F3] rounded p-2 text-white text-xs">
                    Center
                  </button>
                  <button className="bg-[#2a1f4b] border border-gray-600 rounded p-2 text-white text-xs hover:bg-[#3a2f5f]">
                    Center Right
                  </button>
                  <button className="bg-[#2a1f4b] border border-gray-600 rounded p-2 text-white text-xs hover:bg-[#3a2f5f]">
                    Bottom Left
                  </button>
                  <button className="bg-[#2a1f4b] border border-gray-600 rounded p-2 text-white text-xs hover:bg-[#3a2f5f]">
                    Bottom Center
                  </button>
                  <button className="bg-[#2a1f4b] border border-gray-600 rounded p-2 text-white text-xs hover:bg-[#3a2f5f]">
                    Bottom Right
                  </button>
                </div>
              </div>

              {/* Advanced Settings */}
              <div>
                <label className="text-white text-sm font-medium block mb-2">Advanced Settings</label>
                <button className="w-full bg-[#2a1f4b] border border-gray-600 rounded px-3 py-2 text-white text-left hover:bg-[#3a2f5f]">
                  More Options →
                </button>
              </div>
            </div>

            {/* Settings Footer */}
            <div className="p-4 border-t border-gray-600 flex justify-end space-x-2">
              <button
                onClick={() => setIsSettingsOpen(false)}
                className="px-4 py-2 bg-gray-600 text-white rounded hover:bg-gray-700"
              >
                Cancel
              </button>
              <button
                onClick={() => setIsSettingsOpen(false)}
                className="px-4 py-2 bg-[#8356F3] text-white rounded hover:bg-[#7c3aed]"
              >
                Save
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
};
